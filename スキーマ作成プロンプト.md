# Copilotプロンプト: ModelWithTableNameベースクラスを使ったテーブル名とデータ分離構造

## 目標
SQLAlchemyで定義された複数のテーブル（3階層のリレーションを持つ）を、Pydantic v2の`ModelWithTableName`パターンを使って以下の形式のJSONとして返す:

```json
{
  "tables": [
    {
      "table_name": "users",
      "data": {
        "id": 1,
        "username": "john_doe",
        "email": "john@example.com",
        "created_at": "2023-04-15T10:00:00"
      }
    },
    {
      "table_name": "posts",
      "data": {
        "id": 1,
        "title": "初めての投稿",
        "content": "こんにちは、世界！",
        "user_id": 1,
        "created_at": "2023-04-15T10:15:00"
      }
    },
    {
      "table_name": "comments",
      "data": {
        "id": 1,
        "content": "素晴らしい投稿です！",
        "post_id": 1,
        "user_id": 2,
        "created_at": "2023-04-15T10:30:00"
      }
    }
  ]
}
```

from typing import List, Dict, Any, Optional, TypeVar, Generic, Type

# SQLAlchemyのベースクラス定義
class Base(DeclarativeBase):
    pass

# SQLAlchemyモデル定義 - データベーススキーマ
class User(Base):
    __tablename__ = "users"
    
    id = mapped_column(Integer, primary_key=True)
    username = mapped_column(String, unique=True)
    email = mapped_column(String)
    created_at = mapped_column(DateTime, default=datetime.utcnow)
    
    # リレーション
    posts: Mapped[List["Post"]] = relationship("Post", back_populates="user")

class Post(Base):
    __tablename__ = "posts"
    
    id = mapped_column(Integer, primary_key=True)
    title = mapped_column(String)

    user_id = mapped_column(Integer, ForeignKey("users.id"))
